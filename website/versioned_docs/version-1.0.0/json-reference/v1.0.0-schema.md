# v1.0.0 JSON schema

The complete JSON schema for the Native Display SDK. Every field below is documented and stable for v1.0.0.

## Top-level

```ts
type NativeDisplayConfig = {
  theme?:        Theme;
  styleClasses?: StyleClass[];
  variables?:    { [name: string]: any };
  root:          ContainerNode | ElementNode;   // required
};
```

## Containers

```ts
type ContainerNode = {
  type:          "container";
  containerType: "box" | "vertical" | "horizontal" | "gallery";
  id:            string;
  children:      Node[];
  layout?:       Layout;
  style?:        Style;
  styleClass?:   string;
  visible?:      string;             // {{varExpr}}
  actions?:      ActionsBlock;
  animation?:    AnimationBlock;

  // Only when containerType === "gallery":
  galleryConfig?: GalleryConfig;
};
```

Component-specific docs:
- [BOX](/components/containers/box) — overlap layout
- [VERTICAL](/components/containers/vertical) — Column / VStack
- [HORIZONTAL](/components/containers/horizontal) — Row / HStack
- [GALLERY](/components/containers/gallery) — scrolling carousel

## Elements

```ts
type ElementNode = {
  type:        "element";
  elementType: "text" | "image" | "button" | "video" | "html" | "spacer" | "divider";
  id:          string;
  layout?:     Layout;
  style?:      Style;
  styleClass?: string;
  visible?:    string;
  actions?:    ActionsBlock;
  animation?:  AnimationBlock;

  // bindings shape varies by elementType:
  bindings?:     { [key: string]: any };

  // Element-specific config blocks:
  imageConfig?:   { fit?: "crop" | "contain" | "fill" | "tile"; animated?: boolean | null };
  htmlConfig?:    { javascriptEnabled?: boolean; scrollEnabled?: boolean; transparentBackground?: boolean; baseUrl?: string };
  dividerConfig?: { orientation?: "horizontal" | "vertical"; thickness?: number; color?: string };
};
```

| Element | Required bindings |
|---------|-------------------|
| `text` | `text` |
| `image` | `url` |
| `button` | `text` |
| `video` | `url` plus optional `autoPlay`, `loop`, `muted`, `showControls`, `showFullscreen`, `openUrl` |
| `html` | `html` *(takes precedence)* or `url` |
| `spacer`, `divider` | none |

## Layout

```ts
type Layout = {
  width?:       Dimension;
  height?:      Dimension;
  padding?: {
    top?: Dimension; bottom?: Dimension; left?: Dimension; right?: Dimension;
  };
  offset?: {
    x?: Dimension; y?: Dimension;
  };
  aspectRatio?: number;
  arrangement?: ChildArrangement;   // VERTICAL / HORIZONTAL only
};

type ChildArrangement = {
  strategy:    "spaced" | "space_between" | "space_evenly" | "space_around" | "start" | "center" | "end";
  spacing?:    number;     // honoured by `spaced` only
  spacingUnit?: "dp" | "sp" | "px";
};
```

See [layout system](/concepts/layout-system) and [arrangement strategies](/concepts/arrangement-strategies).

## Dimension

```ts
type Dimension =
  | { value: number; unit: "percent" | "dp" | "sp" | "px" }
  | { special: "wrap_content" | "match_parent" };
```

Default if omitted: `{ value: 0, unit: "dp" }`.

## Style

```ts
type Style = {
  // Visual
  backgroundColor?: string;        // "#RRGGBB" or "#RRGGBBAA" or named-colour from theme
  borderRadius?:    Dimension;
  borderWidth?:     number;
  borderColor?:     string;
  shadowColor?:     string;
  shadowRadius?:    number;
  shadowOffsetX?:   number;
  shadowOffsetY?:   number;
  opacity?:         number;        // 0–1

  // Text (cascade)
  textColor?:       string;
  fontSize?:        number | { value: number; unit: "percent" };
  fontFamily?:      string;
  fontWeight?:      "normal" | "medium" | "bold" | "light";
  fontStyle?:       "normal" | "italic";
  lineHeight?:      number | { value: number; unit: "percent" };
  letterSpacing?:   number;
  textDecoration?:  "none" | "underline" | "strikethrough";
  textAlign?:       "start" | "center" | "end" | "justify";
  maxLines?:        number;
  overflow?:        "clip" | "ellipsis" | "visible";
};
```

See [style cascading](/concepts/style-cascading).

## Theme

```ts
type Theme = {
  id?:           string;
  defaultStyle?: Style;
  colors?:       { [name: string]: string };   // "#RRGGBB" or "#RRGGBBAA"
};
```

See [theme](/concepts/theme).

## Style classes

```ts
type StyleClass = { name: string; style: Style };
```

Referenced from a node via `styleClass: "name"`. See [style classes](/concepts/style-classes-deep).

## Animation

```ts
type AnimationBlock = {
  type:     "none" | "fade_in" | "slide_in_left" | "slide_in_right" | "slide_in_top" | "slide_in_bottom"
            | "scale_in" | "fade_scale_in" | "fade_slide_in";
  duration?: number;   // ms
  delay?:    number;   // ms
  easing?:   "linear" | "ease_in" | "ease_out" | "ease_in_out"
            | "ease_in_back" | "ease_out_back" | "spring";
};
```

See [animations](/concepts/animations).

## Actions

```ts
type ActionsBlock = {
  onClick?:      Action[];
  onLongPress?:  Action[];
  onDoubleTap?:  Action[];
  onAppear?:     Action[];
  onDisappear?:  Action[];
};

type Action =
  | { type: "open_url";    url: string }
  | { type: "dismiss" }
  | { type: "track_event"; event: string; properties?: { [k: string]: any } }
  | { type: "custom";      [key: string]: any };
```

See [actions](/concepts/actions).

## Gallery config

```ts
type GalleryConfig = {
  mode?:               "snapping" | "free_flow" | "free_flow_grid";
  orientation?:        "horizontal" | "vertical";
  snapBehavior?:       "none" | "start" | "center" | "end";
  peek?:               { before?: number; after?: number };  // dp
  itemsPerView?:       number;
  spacing?:            number;
  showIndicators?:     boolean;
  autoScrollInterval?: number;   // ms (0 = off)
  infiniteScroll?:     boolean;
  showArrows?:         boolean;
  initialPage?:        number;
};
```

See [GALLERY](/components/containers/gallery).

## Color format

`#RRGGBB` (opaque) or `#RRGGBBAA` (with alpha — RGBA web standard, **not** ARGB). Named colours from the [theme palette](/concepts/theme) work in any colour-accepting field.
